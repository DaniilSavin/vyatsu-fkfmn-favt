// ограничения.
#include <cmath>
#include <stdio.h>

int dif(int h1, int m1, int h2, int m2)
{
    int d = h2*60+m2 - (h1*60+m1); //разница между прилетом и вылетом
    if(d > 60*12)//если разница больше 12 часов
        d -= 60*24; //вычитаем 24 часа
    else if(d < -60*12) //иначе если разница меньше -12 часов
        d += 60*24; //прибавляем 24 часа
    return d; //возвращаем разницу в минутах
}

int main()
{
    int d1, d2;
    int h1, m1, h2, m2;
    scanf("%d.%d %d.%d", &h1, &m1, &h2, &m2);
    d1 = dif(h1, m1, h2, m2);
    scanf("%d.%d %d.%d", &h1, &m1, &h2, &m2);
    d2 = dif(h1, m1, h2, m2);

    for(int d = -5; d <= 5; d++)
    {//если разница прямого рейса минус 300минут >= 0 и разница прямого рейса минус 300минут <=360минут и разница обратного рейса плюс300 минут>=0
        if(d1 - d*60 >= 0 && d1 - d*60 <= 6*60 && d2 + d*60 >= 0 //и разница обратного рейса+300 минут<=360 минут и модуль разницы прилета и отлета меньше 10
           && d2 + d*60 <= 6*60 && std::abs((d1-d*60) - (d2+d*60)) <= 10)
        {
            printf("%d\n", std::abs(d));//выводим модуль разницы
            return 0;
        }
    }
}
