#include <iostream>
#include <math.h>
#include <mpi.h>

int main(int argc, char* argv[])
{
	int proc_rank, proc_num;
	const int buf_size = 100;
	char buf[buf_size];
	MPI_Status st;
	MPI_Init(&argc, &argv);
	MPI_Comm_rank(MPI_COMM_WORLD, &proc_rank);
	MPI_Comm_size(MPI_COMM_WORLD, &proc_num);
	printf_s("I am %d process from %d processes", proc_rank, proc_num);
	MPI_Finalize();
}
